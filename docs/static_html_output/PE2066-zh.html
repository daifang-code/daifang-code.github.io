要求或禁止变量声明语句后有一行空行 (newline-after-var)

(fixable) 命令行中的 --fix 选项可以自动修复一些该规则报告的问题。

该规则在 ESLint v4.0.0 中 已弃用，被 padding-line-between-statements 替代。

目前，变量声明和其余代码如何分开并没有一致性。一些开发者在两者之间保留一行空行：

var foo;

// do something with foo

然而其他人并不这样做。

var foo;
// do something with foo

在一个项目中，这些开发者在一起工作时就成了问题。该规则强制使用一种代码风格，即 var、let 或 const 语句之后是否允许有空行。它有助于整个项目的代码看起来是一致的。

缺陷详因

该规则强制一种代码风格，即在 var、let 或 const 之后要求或禁止有空行，以此在整个项目中达到一致的代码风格。

修改方案

该规则有一个字符串选项：

  • "always" (默认) 要求在 var、let 或 const 之后有空行。声明语句后紧随的注释被当作类似于声明语句
  • "never" 禁止在 var、let 或 const 之后有空行
always

默认选项 "always" 的 错误 代码示例：

/*eslint newline-after-var: ["error", "always"]*/
/*eslint-env es6*/

var greet = "hello,",
    name = "world";
console.log(greet, name);

let greet = "hello,",
    name = "world";
console.log(greet, name);

var greet = "hello,";
const NAME = "world";
console.log(greet, NAME);

var greet = "hello,";
var name = "world";
// var name = require("world");
console.log(greet, name);

默认选项 "always" 的 正确 代码示例：

/*eslint newline-after-var: ["error", "always"]*/
/*eslint-env es6*/

var greet = "hello,",
    name = "world";

console.log(greet, name);

let greet = "hello,",
    name = "world";

console.log(greet, name);

var greet = "hello,";
const NAME = "world";

console.log(greet, NAME);

var greet = "hello,";
var name = "world";
// var name = require("world");

console.log(greet, name);
never

默认选项 "never" 的 错误 代码示例：

/*eslint newline-after-var: ["error", "never"]*/
/*eslint-env es6*/

var greet = "hello,",
    name = "world";

console.log(greet, name);

let greet = "hello,",
    name = "world";

console.log(greet, name);

var greet = "hello,";
const NAME = "world";

console.log(greet, NAME);

var greet = "hello,";
var name = "world";
// var name = require("world");

console.log(greet, name);

选项 "never" 的 正确 代码示例：

/*eslint newline-after-var: ["error", "never"]*/
/*eslint-env es6*/

var greet = "hello,",
    name = "world";
console.log(greet, name);

    name = "world";
