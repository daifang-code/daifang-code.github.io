DUO127

import duo_client

client = duo_client.Client(ikey=..., skey=..., host=..., ca_certs="HTTP")

import duo_client

client = duo_client.Client(ikey=..., skey=..., host=..., ca_certs="DISABLE")

正确代码

import duo_client

client = duo_client.Client(ikey=..., skey=..., host=...)
